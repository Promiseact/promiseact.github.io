<script>
  // Get the modal
  var modal = document.getElementById("myModal");
  
  // Get the <span> element that closes the modal
  var span = document.getElementsByClassName("close")[0];
  
  // When the user clicks on the button, open the modal
  window.onload = function() {
    modal.style.display = "block";
  }
  
  // When the user clicks on <span> (x) or anywhere outside of the modal, close the modal and redirect
  span.onclick = function() {
    modal.style.display = "none";
    window.location.href = "./login.html"; // Change this to your desired redirect page URL
  }
  
  window.onclick = function(event) {
    if (event.target == modal) {
      modal.style.display = "none";
      window.location.href = "./login.html"; // Change this to your desired redirect page URL
    }
  }
  
  // Switch between patient and employee login/register forms
  document.getElementById("patientRegisterLink").onclick = function() {
    document.getElementById("patientLogin").style.display = "none";
    document.getElementById("patientRegister").style.display = "block";
